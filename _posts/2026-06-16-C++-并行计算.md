# 消费者和生产者
```C++

#include <algorithm>
#include <chrono>
#include <iostream>
#include <mutex>
#include <queue>
#include <thread>

#include <ThreadPool/ThreadPool.h>
#include <eventpp/callbacklist.h>
#include <fmt/ranges.h>

class Test {
public:
  Test() : thread_pool_(std::thread::hardware_concurrency() * 2) {
    nodes_.push(0);
  }

  void Run() {
    while (true) {
      int val;
      {
        std::unique_lock<std::mutex> locker(mtx_);
        cv_.wait(locker, [this] { return !nodes_.empty(); });

        val = nodes_.front();
        nodes_.pop();
      }

      if (val > 200)
        break;

      thread_pool_.enqueue([val, this] {
        std::cout << "solving " << val << std::endl;
        std::this_thread::sleep_for(std::chrono::seconds(1));
        mtx_.lock();
        nodes_.push(2 * val + 1);
        nodes_.push(2 * val + 2);
        mtx_.unlock();
        cv_.notify_one();
      });
    }
  }

  ~Test() {}

private:
  std::mutex mtx_;
  std::condition_variable cv_;

  std::queue<int> nodes_;
  ThreadPool thread_pool_;
};

int main() {
  Test test;
  test.Run();

  std::vector<int> vec = {1, 2, 1, 3, 4, 1, 5};

  // 1. std::remove 将非1元素前移，返回新尾迭代器
  auto new_end = std::remove(vec.begin(), vec.end(), 1);
  // 2. erase 截断尾部多余元素
  vec.erase(new_end, vec.end());

  // 遍历输出
  for (int x : vec) {
    std::cout << x << " ";
  }

  auto start = std::chrono::high_resolution_clock::now();
  std::this_thread::sleep_for(std::chrono::seconds(1)); // 睡眠1秒
  auto end = std::chrono::high_resolution_clock::now();
  std::chrono::duration<double> elapsed = end - start;
  std::cout << elapsed.count() << std::endl;

  return 0;
}
```

# 条件变量
```
cv_.wait(locker, [this] { return !nodes_.empty(); });
```
如果`pred`为真，则锁定locker；否则，释放locker，直到cv_.notify_one()/notify_all()
