```C++
class Test {
public:
  Test() : thread_pool_(std::thread::hardware_concurrency() * 2) {
    nodes_.push(0);
  }

  void Run() {
    while (true) {
      int val;
      {
        std::unique_lock<std::mutex> locker(mtx_);
        cv_.wait(locker, [this] { return !nodes_.empty(); });

        val = nodes_.front();
        nodes_.pop();
      }

      if (val > 200)
        break;

      thread_pool_.enqueue([val, this] {
        std::cout << "solving " << val << std::endl;
        std::this_thread::sleep_for(std::chrono::seconds(1));
        mtx_.lock();
        nodes_.push(2 * val + 1);
        nodes_.push(2 * val + 2);
        mtx_.unlock();
        cv_.notify_one();
      });
    }
  }

  ~Test() {}

private:
  std::mutex mtx_;
  std::condition_variable cv_;

  std::queue<int> nodes_;
  ThreadPool thread_pool_;
};
```

# 条件变量
```
cv_.wait(locker, [this] { return !nodes_.empty(); });
```
如果`pred`为真，则锁定locker；否则，释放locker，直到cv_.notify_one()/notify_all()
